```javascript
const prisma = new PrismaClient();

export default async function (app) {
  // –°–æ–∑–¥–∞—Ç—å —Ä–∞—Å—á—ë—Ç
  app.post('/api/calculations', async (request, reply) => {
    const { tariff, items, clientId } = request.body;
    
    const total = items.reduce((sum, item) => 
      sum + (item.volume * tariff * item.quantity), 0
    );
    
    const calculation = await prisma.calculation.create({
      data: {
        tariff,
        total,
        clientId,
        items: {
          create: items.map(item => ({
            fileName: item.fileName,
            volume: item.volume,
            technology: item.technology || 'FDM',
            quantity: item.quantity || 1,
            price: item.volume * tariff
          }))
        }
      },
      include: { items: true }
    });
    
    return calculation;
  });
  
  // –ü–æ–ª—É—á–∏—Ç—å –≤—Å–µ —Ä–∞—Å—á—ë—Ç—ã
  app.get('/api/calculations', async () => {
    return await prisma.calculation.findMany({
      include: { client: true, items: true },
      orderBy: { createdAt: 'desc' }
    });
  });
}
```
